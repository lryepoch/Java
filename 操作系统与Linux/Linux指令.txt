
linux：一切皆文件

·超级权限root用户：#
	一般用户：$

·/：最高级的目录；
	在/底下最常用的3个目录：
		/usr：要使用的应用程序和文件都在这个目录；
		/home：为用户设置的主目录，用~表示；
		/root ：根目录


·常用指令操作【Linux命令严格区分大小写】：

	··文件夹/目录：
		pwd		#查看当前目录完整路径；
		ls -l 		#详细显示当前目录下的所有文件，后面加文件名可以单独显示该文件；
		ls -all	#查看所有文件夹和文件。（包括隐藏的）
		du  		#显示目录或者文件大小
		cd 目录名       #切换目录
		cd ..               #切换到当前目录的上级目录

		mkdir 目录名：创建目录
		mkdir 目录名1 目录2：创建2个同级目录
		mkdir -p 目录名1/目录2：递归创建子目录

		rmdir 目录名：删除目录
		rmdir -p 目录名1/目录2： 递归删除目录，即当子目录被删除时，其父目录为空时也一同被删除。否则只删除子目录。

		find 目录/文件名          #查找指定目录或者文件

	··文件：

		touch 文件名称          #新建空文件
		vi 文件名               	 #打开文件
			其中，点击i，输入文本；
			退出时，先点击esc，再点击：，再输入指令；

			···文件保存：
									：w		             #以默认文件名存盘不退出
								    ：w 	文件名       #以指定文件名存盘不退出
									：wq                 #以默认文件名存盘并退出

									：q		             #不存盘退出
								    ：q！	             #不存盘并强制退出

		echo  写入的内容>文件名称

		cat 文件名                #查看文件内容，注意：这是一种不进入文件中的查看；
		cat -n file1 file2		#一次性查看多个文件，-n参数表示显示行号，从1开始编号；
		more 文件名            #分页形式显示文件全部内容；


		rm 文件名称           #删除文件
		rm –i mydirectory/*   :（删除mydirectory目录下的所有文件，-i表示删前询问）			
		rm -f 文件名           #删除不为空的文件											


		cp：复制文件。    cp myfile* ./mydirectory     （将当前目录下myfile开头的文件复制到当前目录的子目录mydirectory）

		mv：移动文件或者重命名。	mv ../mydirectory/myfile1 student21   （将mydirectory子目录下的myfile1移动到当前目录的子目录student21下）


	··进程：

		ps												#显示当前进程的状态
		ps aux               						 #查看所有进程
		ps aux|grep java 						#查看java进程

		ps –ef|grep tomcat 					#查看所有有关tomcat的进程

		kill 进程号						          #结束进程
		kill -9 进程号  						#强制杀死进程


	··系统级：

			echo string键 				#将string的值输出到屏幕上，说明：方便查看系统环境变量等信息
			who                              #显示目前登入系统的用户信息
			clear								#清屏
			uptime        					#显示系统运行了多长时间
			reboot        					#重启计算机
			halt             					#关闭计算机



----------------------------------
·其他指令：

	··grep 'test' d*　　		#显示所有以d开头的文件中包含 test的行

	··lsof -i：端口号                #用于查看某一端口的占用情况，比如查看22号端口使用情况，lsof -i:22

	··netstat -tunlp|grep 端口号                #用于查看指定端口号的进程情况，如查看22端口的情况，netstat -tunlp|grep 22
